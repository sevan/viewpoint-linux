ncurses
mkdir tmp
cd tmp
configure --prefix=/usr --mandir=/usr/share/man --with-manpage-format=normal --with-shared --without-ada --without-normal --enable-widec --without-debug --libdir=/usr/lib64
make -C include
make -C progs tic

cd ..
configure --prefix=/usr --host=$LFS_TGT --build=$(/viewpoint-linux/lib/ncurses/dist/config.guess) --mandir=/usr/share/man --with-manpage-format=normal --with-shared --without-ada --without-normal --enable-widec --without-debug --libdir=/usr/lib64
make DESTDIR=$LFS TIC_PATH=$(pwd)/tmp/progs/tic install
echo "INPUT(-lncursesw)" > $LFS/usr/lib64/libncurses.so
mv $LFS/usr/lib/terminfo $LFS/usr/lib64/
